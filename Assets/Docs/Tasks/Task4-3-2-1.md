    // In Assets/_Project/Scripts/Data/Collectibles/Talismans/TalismanDatabaseSO.cs
    namespace PetalsOfHope.Data.Collectibles
    {
        using UnityEngine;
        using System.Collections.Generic;
        using System.Linq;

        [CreateAssetMenu(menuName = "Petals of Hope/Data/Collectibles/Talisman Database", fileName = "TalismanDatabaseSO")]
        public class TalismanDatabaseSO : ScriptableObject
        {
            [Tooltip("List of all talisman data assets in the game. Populate this in the editor.")]
            public List<TalismanDataSO> allTalismans;

            private Dictionary<string, TalismanDataSO> _talismanMap;
            
            // Public static instance for easy access, assuming only one such database SO exists.
            // This SO would need to be preloaded or assigned to systems that need it.
            // A more robust approach might involve Resources.Load or Addressables.
            private static TalismanDatabaseSO _instance;
            public static TalismanDatabaseSO Instance 
            {
                get 
                {
                    if (_instance == null)
                    {
                        // Attempt to load from a predefined path in Resources if not found.
                        // This requires the SO asset to be placed in a Resources folder.
                        _instance = Resources.Load<TalismanDatabaseSO>("Data/TalismanDatabaseSO"); 
                        if (_instance == null) {
                            Debug.LogError("TalismanDatabaseSO instance could not be loaded from Resources/Data/TalismanDatabaseSO. Ensure it exists and is correctly named/pathed.");
                        } else {
                            _instance.InitializeMap(); // Initialize map when loaded
                        }
                    }
                    return _instance;
                }
            }


            private void OnEnable() // Called when SO is loaded
            {
                 // Initialize map if instance is this one, or if _instance is null (first load)
                if (_instance == null || _instance == this) {
                    _instance = this; // In case it's loaded from being referenced directly, not via Resources.Load
                    InitializeMap();
                }
            }

            private void InitializeMap()
            {
                if (allTalismans == null)
                {
                    Debug.LogError("TalismanDatabaseSO: allTalismans list is null!", this);
                    _talismanMap = new Dictionary<string, TalismanDataSO>();
                    return;
                }

                // Filter out null entries and talismans with empty IDs before creating the map.
                _talismanMap = allTalismans
                    .Where(t => t != null && !string.IsNullOrEmpty(t.talismanID))
                    .GroupBy(t => t.talismanID) // Group by ID
                    .ToDictionary(g => g.Key, g => { // Handle duplicates
                        if (g.Count() > 1) {
                            Debug.LogWarning($"Duplicate talismanID '{g.Key}' found in TalismanDatabaseSO. Using the first one.", this);
                        }
                        return g.First();
                    });
                
                if (_talismanMap.Count != allTalismans.Count(t => t != null && !string.IsNullOrEmpty(t.talismanID))) {
                    Debug.LogWarning("Some talismans were not added to the map due to missing ID or being null.", this);
                }
            }

            public TalismanDataSO GetTalismanByID(string id)
            {
                if (_talismanMap == null) {
                    InitializeMap(); // Ensure map is initialized if accessed before OnEnable/Instance getter
                }

                if (string.IsNullOrEmpty(id) || _talismanMap == null) return null;
                _talismanMap.TryGetValue(id, out TalismanDataSO talisman);
                if (talisman == null) {
                    // Debug.LogWarning($"Talisman with ID '{id}' not found in database.", this);
                }
                return talisman;
            }
        }
    }
